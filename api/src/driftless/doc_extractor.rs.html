10/// Extracted documentation for a task type
11#[derive(Debug, Clone)]
12pub struct TaskDocumentation {
13    /// Human-readable description
14    pub description: String,
15    /// Field documentation
16    pub fields: HashMap<String, FieldDocumentation>,
17    /// Examples from the implementation file
18    pub examples: Vec<TaskExample>,
19    /// Registered output documentation
20    pub register_outputs: HashMap<String, RegisterOutputDocumentation>,
21}

23/// Documentation for a task field
24#[derive(Debug, Clone)]
25pub struct FieldDocumentation {

28    /// Field type description
29    pub field_type: String,
30    /// Whether the field is required
31    pub required: bool,
32    /// Human-readable description
33    pub description: String,
34}

62/// Extract documentation for all tasks
63pub fn extract_all_task_docs() -> Result<HashMap<String, TaskDocumentation>> {
64    let mut docs = HashMap::new();
65
66    // Get all registered task types from the registry
67    let registered_task_types = crate::apply::TaskRegistry::get_registered_task_types();
68
69    // Extract from each task file
70    for task_type in registered_task_types {
71        let filename = crate::apply::TaskAction::task_filename(&task_type);
72        let file_path = format!("src/apply/{}.rs", filename);
73        if let Ok(content) = fs::read_to_string(&file_path) {
74            extract_task_struct_docs(&content, &mut docs, &task_type)?;
75        }
76    }
77
78    // Extract examples from implementation files
79    extract_examples_from_files(&mut docs)?;
80
81    // Add common fields to all tasks
82    add_common_task_fields(&mut docs);
83
84    Ok(docs)
85}

87/// Add common fields (register, when) to all task documentation
88fn add_common_task_fields(docs: &mut HashMap<String, TaskDocumentation>) {
89    for task_doc in docs.values_mut() {
90        // Always add 'when' field documentation
91        task_doc.fields.insert(
92            "when".to_string(),
93            FieldDocumentation {
94                name: "when".to_string(),
95                field_type: "Option<String>".to_string(),
96                required: false,
97                description: "Optional condition to determine if the task should run".to_string(),
98            },
99        );
100
101        // Only add 'register' field documentation if the task has outputs to register
102        if !task_doc.register_outputs.is_empty() {
103            task_doc.fields.insert(
104                "register".to_string(),
105                FieldDocumentation {
106                    name: "register".to_string(),
107                    field_type: "Option<String>".to_string(),
108                    required: false,
109                    description: "Optional variable name to register the task result in"
110                        .to_string(),
111                },
112            );
113        }
114    }
115}

161/// Extract task struct documentation from mod.rs
162fn extract_task_struct_docs(
163    content: &str,
164    docs: &mut HashMap<String, TaskDocumentation>,
165    task_type: &str,
166) -> Result<()> {
167    let lines: Vec<&str> = content.lines().collect();
168    let mut i = 0;
169
170    while i < lines.len() {
171        // Look for task struct definitions
172        if lines[i].contains("#[derive") && lines[i + 1].contains("pub struct") {
173            let struct_line = lines[i + 1];
174            if let Some(struct_name) = extract_struct_name(struct_line) {
175                // If it ends with Task, assume it's the main task struct for this file
176                if struct_name.ends_with("Task") {
177                    let mut task_doc = TaskDocumentation {
178                        description: String::new(),
179                        fields: HashMap::new(),
180                        examples: Vec::new(),
181                        register_outputs: HashMap::new(),
182                    };
183
184                    // Extract struct documentation (go backwards to find doc comments)
185                    let mut doc_lines = Vec::new();
186                    let mut j = i;
187                    while j > 0 {
188                        j -= 1;
189                        let line = lines[j].trim();
190                        if line.starts_with("///") {
191                            let content = line.trim_start_matches("///").trim();
192                            doc_lines.insert(0, content.to_string());
193                        } else if !line.is_empty()
194                            && !line.starts_with("//")
195                            && !line.starts_with("#[")
196                        {
197                            break;
198                        }
199                    }
200
201                    if !doc_lines.is_empty() {
202                        // Separate description from register outputs
203                        let mut description_lines = Vec::new();
204                        let mut in_outputs_section = false;
205
206                        for line in doc_lines {
207                            let trimmed = line.trim();
208                            if trimmed == "# Registered Outputs" {
209                                in_outputs_section = true;
210                                continue;
211                            }
212
213                            if in_outputs_section {
214                                if trimmed.starts_with("- `") {
215                                    if let Some(type_start) = trimmed.find("` (") {
216                                        if let Some(type_end) = trimmed.find("): ") {
217                                            let name = trimmed[3..type_start].to_string();
218                                            let output_type =
219                                                trimmed[type_start + 3..type_end].to_string();
220                                            let desc = trimmed[type_end + 3..].to_string();
221                                            task_doc.register_outputs.insert(
222                                                name.clone(),
223                                                RegisterOutputDocumentation {
224                                                    name,
225                                                    output_type,
226                                                    description: desc,
227                                                },
228                                            );
229                                        }
230                                    } else if let Some(pos) = trimmed.find("`: ") {
231                                        let name = trimmed[3..pos].to_string();
232                                        let desc = trimmed[pos + 3..].to_string();
233                                        task_doc.register_outputs.insert(
234                                            name.clone(),
235                                            RegisterOutputDocumentation {
236                                                name,
237                                                output_type: "Unknown".to_string(),
238                                                description: desc,
239                                            },
240                                        );
241                                    }
242                                }
243                            } else {
244                                description_lines.push(line);
245                            }
246                        }
247                        task_doc.description = description_lines.join("\n").trim().to_string();
248                    }
249
250                    // Extract field documentation
251                    i += 2; // Skip the derive and struct lines
252                    while i < lines.len() && !lines[i].contains('}') {
253                        if lines[i].contains("pub ") && lines[i].contains(": ") {
254                            if let Some(field_doc) = extract_field_doc(&lines, &mut i) {
255                                task_doc.fields.insert(field_doc.name.clone(), field_doc);
256                            }
257                        } else {
258                            i += 1;
259                        }
260                    }
261
262                    docs.insert(task_type.to_string(), task_doc);
263                }
264            }
265        }
266        i += 1;
267    }
268
269    Ok(())
270}

272/// Extract struct name from a struct definition line
273fn extract_struct_name(line: &str) -> Option<String> {
274    let parts: Vec<&str> = line.split_whitespace().collect();
275    if parts.len() >= 3 && parts[0] == "pub" && parts[1] == "struct" {
276        Some(parts[2].to_string())
277    } else {
278        None
279    }
280}

282/// Extract field documentation
283fn extract_field_doc(lines: &[&str], i: &mut usize) -> Option<FieldDocumentation> {
284    let field_line = lines[*i];
285    let field_name = extract_field_name(field_line)?;
286
287    // Extract field documentation (go backwards to find doc comments)
288    let mut doc_lines = Vec::new();
289    let mut j = *i;
290    while j > 0 {
291        j -= 1;
292        let line = lines[j].trim();
293        if line.starts_with("///") {
294            doc_lines.insert(0, line.trim_start_matches("///").trim());
295        } else if line.starts_with("#[") || line.is_empty() {
296            // Skip attributes and empty lines, continue looking for docs
297            continue;
298        } else if !line.is_empty() && !line.starts_with("//") {
299            // Stop at actual code that's not an attribute
300            break;
301        }
302    }
303
304    let description = if doc_lines.is_empty() {
305        "No description available".to_string()
306    } else {
307        doc_lines.join("\n")
308    };
309
310    let required = !field_line.contains("Option<");
311    let field_type = extract_field_type(field_line)?;
312
313    *i += 1;
314
315    Some(FieldDocumentation {
316        name: field_name,
317        field_type,
318        required,
319        description,
320    })
321}

323/// Extract field name from a field definition line
324fn extract_field_name(line: &str) -> Option<String> {
325    let line = line.trim();
326    line.find("pub ").and_then(|pub_pos| {
327        let after_pub = &line[pub_pos + 4..];
328        after_pub
329            .find(": ")
330            .map(|colon_pos| after_pub[..colon_pos].trim().to_string())
331    })
332}

334/// Extract field type from a field definition line
335fn extract_field_type(line: &str) -> Option<String> {
336    let line = line.trim();
337    if let Some(colon_pos) = line.find(": ") {
338        let type_part = &line[colon_pos + 2..];
339
340        // Find the comma that ends this field (not commas inside generics)
341        let mut paren_depth = 0;
342        let mut bracket_depth = 0;
343        let mut angle_depth = 0;
344
345        for (i, c) in type_part.char_indices() {
346            match c {
347                '(' => paren_depth += 1,
348                ')' => paren_depth -= 1,
349                '[' => bracket_depth += 1,
350                ']' => bracket_depth -= 1,
351                '<' => angle_depth += 1,
352                '>' => angle_depth -= 1,
353                ',' => {
354                    // This is the field-ending comma if we're at depth 0
355                    if paren_depth == 0 && bracket_depth == 0 && angle_depth == 0 {
356                        let type_str = type_part[..i].trim().to_string();
357                        return Some(simplify_type_name(&type_str));
358                    }
359                }
360                _ => {}
361            }
362        }
363
364        // No comma found, take everything until end (last field in struct)
365        let type_str = type_part.trim().trim_end_matches(',').trim().to_string();
366        Some(simplify_type_name(&type_str))
367    } else {
368        None
369    }
370}

372/// Simplify type names by removing std:: prefixes and common simplifications
373fn simplify_type_name(type_name: &str) -> String {
374    let simplified = type_name
375        .replace("std::collections::", "")
376        .replace("std::option::", "")
377        .replace("std::vec::", "")
378        .replace("serde_yaml::", "")
379        .replace("serde_json::", "");
380
381    // Handle Option<T> -> Option<T> (keep as is, but simplify inner types)
382    if simplified.starts_with("Option<") && simplified.ends_with('>') {
383        let inner = &simplified[7..simplified.len() - 1];
384        format!("Option<{}>", simplify_type_name(inner))
385    } else if simplified.starts_with("Vec<") && simplified.ends_with('>') {
386        let inner = &simplified[4..simplified.len() - 1];
387        format!("Vec<{}>", simplify_type_name(inner))
388    } else if simplified.starts_with("HashMap<") && simplified.ends_with('>') {
389        let inner = &simplified[8..simplified.len() - 1];
390        format!("HashMap<{}>", simplify_type_name(inner))
391    } else {
392        simplified
393    }

395fn extract_examples_from_files(docs: &mut HashMap<String, TaskDocumentation>) -> Result<()> {
396    // Get all registered task types from the registry
397    let registered_task_types = crate::apply::TaskRegistry::get_registered_task_types();
398
399    // Generate file paths from task types using the filename mapping
400    let task_files: Vec<(String, String)> = registered_task_types
401        .iter()
402        .map(|task_type| {
403            let filename = crate::apply::TaskAction::task_filename(task_type);
404            let file_path = format!("src/apply/{}.rs", filename);
405            (task_type.clone(), file_path)
406        })
407        .collect();
408
409    for (task_type, file_path) in task_files {
410        if let Ok(content) = fs::read_to_string(&file_path) {
411            if let Some(examples) = extract_examples_from_file(&content) {
412                if let Some(task_doc) = docs.get_mut(&task_type) {
413                    task_doc.examples = examples;
414                }
415            }
416        }
417    }
418
419    Ok(())
420}

422/// Extract examples from a single implementation file
423fn extract_examples_from_file(content: &str) -> Option<Vec<TaskExample>> {
424    let mut examples = Vec::new();
425    let lines: Vec<&str> = content.lines().collect();
426
427    let mut i = 0;
428    while i < lines.len() {
429        let mut found_example = false;
430        let mut description = String::new();
431
432        // Look for example sections starting with //! ##
433        if lines[i].starts_with("//! ##") && !lines[i].starts_with("//! ###") {
434            description = lines[i]
435                .trim_start_matches("//! ##")
436                .trim_start()
437                .to_string();
438            found_example = true;
439            i += 1; // Move to next line
440        }
441
442        if found_example {
443            // Skip description lines until we find **YAML Format:** or another header
444            while i < lines.len()
445                && !lines[i].contains("**YAML Format:**")
446                && !lines[i].contains("## ")
447            {
448                i += 1;
449            }
450
451            if i >= lines.len() || lines[i].contains("## ") {
452                continue;
453            }
454
455            i += 1; // Skip the **YAML Format:** line
456
457            // Skip the opening ```yaml line
458            if i < lines.len() && lines[i].starts_with("//! ```") {
459                i += 1;
460            }
461
462            // Extract YAML block
463            let mut yaml_lines = Vec::new();
464            let mut min_indent = usize::MAX;
465
466            // First pass: find minimum indentation
467            let start_i = i;
468            while i < lines.len() && !lines[i].starts_with("//! ```") {
469                if lines[i].starts_with("//!") {
470                    let line = lines[i].trim_start_matches("//!").trim_end();
471                    if !line.is_empty() {
472                        let indent = line.len() - line.trim_start().len();
473                        min_indent = min_indent.min(indent);
474                    }
475                }
476                i += 1;
477            }
478
479            // Reset to start and extract with proper indentation
480            i = start_i;
481            while i < lines.len() && !lines[i].starts_with("//! ```") {
482                if lines[i].starts_with("//!") {
483                    let line = lines[i].trim_start_matches("//!").trim_end();
484                    if !line.is_empty() || yaml_lines.is_empty() {
485                        // Remove the minimum indentation to preserve relative structure
486                        let trimmed_line = if line.len() >= min_indent {
487                            &line[min_indent..]
488                        } else {
489                            line.trim_start()
490                        };
491                        yaml_lines.push(trimmed_line.to_string());
492                    }
493                }
494                i += 1;
495            }
496            i += 1; // Skip the closing ```
497
498            // Skip to **JSON Format:** or another header
499            while i < lines.len()
500                && !lines[i].contains("**JSON Format:**")
501                && !lines[i].contains("## ")
502            {
503                i += 1;
504            }
505
506            if i >= lines.len() || lines[i].contains("## ") {
507                continue;
508            }
509
510            i += 1; // Skip the **JSON Format:** line
511
512            // Skip the opening ```json line
513            if i < lines.len() && lines[i].starts_with("//! ```") {
514                i += 1;
515            }
516
517            // Extract JSON block
518            let mut json_lines = Vec::new();
519            let mut min_indent = usize::MAX;
520
521            // First pass: find minimum indentation
522            let start_i = i;
523            while i < lines.len() && !lines[i].starts_with("//! ```") {
524                if lines[i].starts_with("//!") {
525                    let line = lines[i].trim_start_matches("//!").trim_end();
526                    if !line.is_empty() {
527                        let indent = line.len() - line.trim_start().len();
528                        min_indent = min_indent.min(indent);
529                    }
530                }
531                i += 1;
532            }
533
534            // Reset to start and extract with proper indentation
535            i = start_i;
536            while i < lines.len() && !lines[i].starts_with("//! ```") {
537                if lines[i].starts_with("//!") {
538                    let line = lines[i].trim_start_matches("//!").trim_end();
539                    if !line.is_empty() || json_lines.is_empty() {
540                        // Remove the minimum indentation to preserve relative structure
541                        let trimmed_line = if line.len() >= min_indent {
542                            &line[min_indent..]
543                        } else {
544                            line.trim_start()
545                        };
546                        json_lines.push(trimmed_line.to_string());
547                    }
548                }
549                i += 1;
550            }
551            i += 1; // Skip the closing ```
552
553            // Skip to **TOML Format:** or another header
554            while i < lines.len()
555                && !lines[i].contains("**TOML Format:**")
556                && !lines[i].contains("## ")
557            {
558                i += 1;
559            }
560
561            if i >= lines.len() || lines[i].contains("## ") {
562                continue;
563            }
564
565            i += 1; // Skip the **TOML Format:** line
566
567            // Skip the opening ```toml line
568            if i < lines.len() && lines[i].starts_with("//! ```") {
569                i += 1;
570            }
571
572            // Extract TOML block
573            let mut toml_lines = Vec::new();
574            let mut min_indent = usize::MAX;
575
576            // First pass: find minimum indentation
577            let start_i = i;
578            while i < lines.len() && !lines[i].starts_with("//! ```") {
579                if lines[i].starts_with("//!") {
580                    let line = lines[i].trim_start_matches("//!").trim_end();
581                    if !line.is_empty() {
582                        let indent = line.len() - line.trim_start().len();
583                        min_indent = min_indent.min(indent);
584                    }
585                }
586                i += 1;
587            }
588
589            // Reset to start and extract with proper indentation
590            i = start_i;
591            while i < lines.len() && !lines[i].starts_with("//! ```") {
592                if lines[i].starts_with("//!") {
593                    let line = lines[i].trim_start_matches("//!").trim_end();
594                    if !line.is_empty() || toml_lines.is_empty() {
595                        // Remove the minimum indentation to preserve relative structure
596                        let trimmed_line = if line.len() >= min_indent {
597                            &line[min_indent..]
598                        } else {
599                            line.trim_start()
600                        };
601                        toml_lines.push(trimmed_line.to_string());
602                    }
603                }
604                i += 1;
605            }
606            i += 1; // Skip the closing ```
607
608            // Skip to **Output:** or another header
609            while i < lines.len()
610                && !lines[i].contains("**Output:**")
611                && !lines[i].contains("//! ## ")
612            {
613                i += 1;
614            }
615
616            let mut output_lines = Vec::new();
617            if i < lines.len() && lines[i].contains("**Output:**") {
618                i += 1; // Skip the **Output:** line
619
620                // Skip the opening ```yaml line
621                if i < lines.len() && lines[i].starts_with("//! ```") {
622                    i += 1;
623                }
624
625                // Extract output block
626                while i < lines.len() && !lines[i].starts_with("//! ```") {
627                    if lines[i].starts_with("//!") {
628                        let line = lines[i].trim_start_matches("//!").trim_end();
629                        if !line.is_empty() || output_lines.is_empty() {
630                            output_lines.push(line.to_string());
631                        }
632                    }
633                    i += 1;
634                }
635                i += 1; // Skip the closing ```
636            }
637
638            let yaml = yaml_lines.join("\n");
639            let json = json_lines.join("\n");
640            let toml = toml_lines.join("\n");
641            let output = output_lines.join("\n");
642
643            if !yaml.is_empty() && !json.is_empty() && !toml.is_empty() {
644                examples.push(TaskExample {
645                    description,
646                    yaml,
647                    json,
648                    toml,
649                    output,
650                });
651            }
652
653            // Skip to the next example or end of file
654            while i < lines.len() && !lines[i].starts_with("//! ##") {
655                i += 1;
656            }
657        } else {
658            i += 1;
659        }
660    }
661
662    if examples.is_empty() {
663        None
664    } else {
665        Some(examples)
666    }
667}

669/// Extract examples from facts implementation files
670fn extract_facts_examples_from_files(docs: &mut HashMap<String, TaskDocumentation>) -> Result<()> {
671    // Get all registered collector types from the registry
672    let registered_collector_types = crate::facts::FactsRegistry::get_registered_collector_types();
673
674    // Generate file paths from collector types using the filename mapping
675    let collector_files: Vec<(String, String)> = registered_collector_types
676        .iter()
677        .map(|collector_type| {
678            let filename = crate::facts::FactsRegistry::get_collector_filename(collector_type);
679            let file_path = format!("src/facts/{}.rs", filename);
680            (collector_type.clone(), file_path)
681        })
682        .collect();
683
684    for (collector_type, file_path) in collector_files {
685        if let Ok(content) = fs::read_to_string(&file_path) {
686            if let Some(examples) = extract_examples_from_file(&content) {
687                if let Some(collector_doc) = docs.get_mut(&collector_type) {
688                    collector_doc.examples = examples;
689                }
690            }
691        }
692    }
693
694    Ok(())
695}

703    let file_path = "src/logs/mod.rs";
704    if let Ok(content) = fs::read_to_string(file_path) {
705        if let Some(examples) = extract_examples_from_file(&content) {
706            // Since logs examples are not specific to processor types yet,
707            // we'll add them to all processor docs for now
708            for processor_doc in docs.values_mut() {
709                processor_doc.examples = examples.clone();
710            }
711        }
712    }
713
714    Ok(())
715}

717/// Extract facts collector documentation from mod.rs
718fn extract_facts_struct_docs(
719    content: &str,
720    docs: &mut HashMap<String, TaskDocumentation>,
721    collector_types: &[String],
722) -> Result<()> {
723    let lines: Vec<&str> = content.lines().collect();
724    let mut i = 0;
725
726    while i < lines.len() {
727        // Look for collector struct definitions
728        if lines[i].contains("#[derive") && lines[i + 1].contains("pub struct") {
729            let struct_line = lines[i + 1];
730            if let Some(struct_name) = extract_struct_name(struct_line) {
731                // If it ends with Collector, assume it's a collector struct
732                if struct_name.ends_with("Collector") {
733                    let collector_type = struct_name.trim_end_matches("Collector").to_lowercase();
734                    if collector_types.contains(&collector_type) {
735                        let description =
736                            crate::facts::FactsRegistry::get_collector_description(&collector_type);
737                        let mut task_doc = TaskDocumentation {
738                            description,
739                            fields: HashMap::new(),
740                            examples: Vec::new(),
741                            register_outputs: HashMap::new(),
742                        };
743
744                        // Extract field documentation
745                        extract_struct_fields(&lines, &mut i, &mut task_doc.fields)?;
746
747                        docs.insert(collector_type, task_doc);
748                    }
749                }
750            }
751        }
752        i += 1;
753    }
754
755    Ok(())
756}

758/// Extract all fields from a struct
759fn extract_struct_fields(
760    lines: &[&str],
761    i: &mut usize,
762    fields: &mut HashMap<String, FieldDocumentation>,
763) -> Result<()> {
764    // Skip the derive and struct lines
765    *i += 2;
766
767    while *i < lines.len() && !lines[*i].contains('}') {
768        if lines[*i].contains("pub ") && lines[*i].contains(": ") {
769            if let Some(field_doc) = extract_field_doc(lines, i) {
770                fields.insert(field_doc.name.clone(), field_doc);
771            }
772        } else {
773            *i += 1;
774        }
775    }
776
777    Ok(())
778}
