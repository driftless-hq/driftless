1//! Disk facts collector
2//!
3//! Collects disk space and I/O statistics for mounted filesystems.
4//!
5//! # Examples
6//!
7//! ## Basic disk metrics collection
8//!
9//! **YAML Format:**
10//! ```yaml
11//! type: disk
12//! name: disk
13//! devices: ["/dev/sda", "/dev/sdb"]
14//! mount_points: ["/", "/home", "/var"]
15//! collect:
16//!   total: true
17//!   used: true
18//!   free: true
19//!   available: true
20//!   percentage: true
21//!   io: true
22//! thresholds:
23//!   usage_warning: 80.0
24//!   usage_critical: 90.0
25//! ```
26//!
27//! **JSON Format:**
28//! ```json
29//! {
30//!   "type": "disk",
31//!   "name": "disk",
32//!   "devices": ["/dev/sda", "/dev/sdb"],
33//!   "mount_points": ["/", "/home", "/var"],
34//!   "collect": {
35//!     "total": true,
36//!     "used": true,
37//!     "free": true,
38//!     "available": true,
39//!     "percentage": true,
40//!     "io": true
41//!   },
42//!   "thresholds": {
43//!     "usage_warning": 80.0,
44//!     "usage_critical": 90.0
45//!   }
46//! }
47//! ```
48//!
49//! **TOML Format:**
50//! ```toml
51//! [[collectors]]
52//! type = "disk"
53//! name = "disk"
54//! devices = ["/dev/sda", "/dev/sdb"]
55//! mount_points = ["/", "/home", "/var"]
56//!
57//! [collectors.collect]
58//! total = true
59//! used = true
60//! free = true
61//! available = true
62//! percentage = true
63//! io = true
64//!
65//! [collectors.thresholds]
66//! usage_warning = 80.0
67//! usage_critical = 90.0
68//! ```
69//!
70//! **Output:**
71//! ```yaml
72//! disks:
73//!   - device: "/dev/sda1"
74//!     mount_point: "/"
75//!     is_removable: false
76//!     total_bytes: 536870912000
77//!     total_mb: 512000
78//!     total_gb: 500
79//!     used_bytes: 268435456000
80//!     used_mb: 256000
81//!     used_gb: 250
82//!     free_bytes: 134217728000
83//!     free_mb: 128000
84//!     free_gb: 125
85//!     available_bytes: 107374182400
86//!     available_mb: 102400
87//!     available_gb: 100
88//!     usage_percent: 50
89//!     available_percent: 20
90//!     disk_pressure: "medium"
91//!     usage_warning: false
92//!     usage_critical: false
93//!     io_supported: false
94//! labels:
95//!   storage_type: ssd
96//! ```

98use crate::facts::DiskCollector;
99use anyhow::Result;
100use serde_yaml::Value;
101use std::collections::HashMap;
102use sysinfo::{Disks, System};
103
104/// Execute disk facts collection
105pub fn collect_disk_facts(collector: &DiskCollector) -> Result<Value> {
106    let mut system = System::new();
107    system.refresh_all();
108
109    let mut disks = Disks::new();
110    disks.refresh();
111
112    let mut facts = HashMap::new();
113    let mut disks_info = Vec::new();
114
115    // Iterate over all disks
116    for disk in disks.list() {
117        let disk_name = disk.name().to_string_lossy().to_string();
118        let mount_point = disk.mount_point().to_string_lossy().to_string();
119
120        // Filter by devices if specified
121        if !collector.devices.is_empty() && !collector.devices.contains(&disk_name) {
122            continue;
123        }
124
125        // Filter by mount points if specified
126        if !collector.mount_points.is_empty() && !collector.mount_points.contains(&mount_point) {
127            continue;
128        }
129
130        let mut disk_info = HashMap::new();
131
132        // Basic disk info
133        disk_info.insert("device".to_string(), Value::String(disk_name.clone()));
134        disk_info.insert(
135            "mount_point".to_string(),
136            Value::String(mount_point.clone()),
137        );
138        disk_info.insert("is_removable".to_string(), Value::Bool(disk.is_removable()));
139
140        let total_space = disk.total_space();
141        let available_space = disk.available_space();
142        let used_space = total_space.saturating_sub(available_space);
143
144        // Collect total space
145        if collector.collect.total {
146            disk_info.insert("total_bytes".to_string(), Value::Number(total_space.into()));
147            disk_info.insert(
148                "total_mb".to_string(),
149                Value::Number((total_space / 1024 / 1024).into()),
150            );
151            disk_info.insert(
152                "total_gb".to_string(),
153                Value::Number((total_space / 1024 / 1024 / 1024).into()),
154            );
155        }
156
157        // Collect used space
158        if collector.collect.used {
159            disk_info.insert("used_bytes".to_string(), Value::Number(used_space.into()));
160            disk_info.insert(
161                "used_mb".to_string(),
162                Value::Number((used_space / 1024 / 1024).into()),
163            );
164            disk_info.insert(
165                "used_gb".to_string(),
166                Value::Number((used_space / 1024 / 1024 / 1024).into()),
167            );
168        }
169
170        // Collect free space (total - used)
171        if collector.collect.free {
172            let free_space = total_space.saturating_sub(used_space);
173            disk_info.insert("free_bytes".to_string(), Value::Number(free_space.into()));
174            disk_info.insert(
175                "free_mb".to_string(),
176                Value::Number((free_space / 1024 / 1024).into()),
177            );
178            disk_info.insert(
179                "free_gb".to_string(),
180                Value::Number((free_space / 1024 / 1024 / 1024).into()),
181            );
182        }
183
184        // Collect available space
185        if collector.collect.available {
186            disk_info.insert(
187                "available_bytes".to_string(),
188                Value::Number(available_space.into()),
189            );
190            disk_info.insert(
191                "available_mb".to_string(),
192                Value::Number((available_space / 1024 / 1024).into()),
193            );
194            disk_info.insert(
195                "available_gb".to_string(),
196                Value::Number((available_space / 1024 / 1024 / 1024).into()),
197            );
198        }
199
200        // Collect usage percentage and pressure monitoring
201        if collector.collect.percentage {
202            let usage_percent = if total_space > 0 {
203                (used_space as f64 / total_space as f64) * 100.0
204            } else {
205                0.0
206            };
207            disk_info.insert(
208                "usage_percent".to_string(),
209                Value::Number((usage_percent as i64).into()),
210            );
211
212            let available_percent = if total_space > 0 {
213                (available_space as f64 / total_space as f64) * 100.0
214            } else {
215                0.0
216            };
217            disk_info.insert(
218                "available_percent".to_string(),
219                Value::Number((available_percent as i64).into()),
220            );
221
222            // Disk pressure monitoring based on available space
223            let disk_pressure = if available_percent < 5.0 {
224                "critical"
225            } else if available_percent < 10.0 {
226                "high"
227            } else if available_percent < 20.0 {
228                "medium"
229            } else {
230                "low"
231            };
232            disk_info.insert(
233                "disk_pressure".to_string(),
234                Value::String(disk_pressure.to_string()),
235            );
236
237            // Check thresholds
238            if let Some(warning) = collector.thresholds.usage_warning {
239                disk_info.insert(
240                    "usage_warning".to_string(),
241                    Value::Bool(usage_percent >= warning),
242                );
243            }
244            if let Some(critical) = collector.thresholds.usage_critical {
245                disk_info.insert(
246                    "usage_critical".to_string(),
247                    Value::Bool(usage_percent >= critical),
248                );
249            }
250        }
251
252        // Collect I/O statistics if available
253        if collector.collect.io {
254            // Note: sysinfo 0.30 may not have I/O stats, this is a placeholder
255            // In future versions, this might include read/write bytes, operations, etc.
256            disk_info.insert("io_supported".to_string(), Value::Bool(false));
257        }
258
259        disks_info.push(Value::Mapping(
260            disk_info
261                .into_iter()
262                .map(|(k, v)| (Value::String(k), v))
263                .collect(),
264        ));
265    }
266
267    facts.insert("disks".to_string(), Value::Sequence(disks_info));
268
269    // Add base labels if any
270    if !collector.base.labels.is_empty() {
271        let mut labels = HashMap::new();
272        for (key, value) in &collector.base.labels {
273            labels.insert(key.clone(), Value::String(value.clone()));
274        }
275        facts.insert(
276            "labels".to_string(),
277            Value::Mapping(
278                labels
279                    .into_iter()
280                    .map(|(k, v)| (Value::String(k), v))
281                    .collect(),
282            ),
283        );
284    }
285
286    Ok(Value::Mapping(
287        facts
288            .into_iter()
289            .map(|(k, v)| (Value::String(k), v))
290            .collect(),
291    ))
292}

300    #[test]
301    fn test_collect_disk_facts_basic() {
302        let collector = DiskCollector {
303            base: BaseCollector {
304                name: "disk".to_string(),
305                enabled: true,
306                poll_interval: 60,
307                labels: HashMap::new(),
308            },
309            devices: vec!["/dev/sda".to_string()],
310            mount_points: vec!["/".to_string(), "/home".to_string()],
311            collect: DiskCollectOptions {
312                total: true,
313                used: true,
314                free: true,
315                available: true,
316                percentage: true,
317                io: true,
318            },
319            thresholds: DiskThresholds {
320                usage_warning: Some(80.0),
321                usage_critical: Some(90.0),
322            },
323        };
324
325        let result = collect_disk_facts(&collector);
326        assert!(result.is_ok());
327
328        let value = result.unwrap();
329        if let Value::Mapping(map) = value {
330            let keys: std::collections::HashSet<_> = map
331                .keys()
332                .filter_map(|k| {
333                    if let Value::String(s) = k {
334                        Some(s.as_str())
335                    } else {
336                        None
337                    }
338                })
339                .collect();
340
341            // Should have disks array
342            assert!(keys.contains("disks"));
343
344            // Check disks array
345            if let Some(Value::Sequence(disks)) = map.get(&Value::String("disks".to_string())) {
346                // Should have at least one disk (depending on system)
347                // Note: In test environment, this might be empty, so we don't assert length
348
349                // If there are disks, validate structure
350                for disk in disks {
351                    if let Value::Mapping(disk_map) = disk {
352                        let disk_keys: std::collections::HashSet<_> = disk_map
353                            .keys()
354                            .filter_map(|k| {
355                                if let Value::String(s) = k {
356                                    Some(s.as_str())
357                                } else {
358                                    None
359                                }
360                            })
361                            .collect();
362
363                        // Should have basic fields
364                        assert!(disk_keys.contains("device"));
365                        assert!(disk_keys.contains("mount_point"));
366                        assert!(disk_keys.contains("is_removable"));
367
368                        // Should have space metrics
369                        assert!(disk_keys.contains("total_bytes"));
370                        assert!(disk_keys.contains("used_bytes"));
371                        assert!(disk_keys.contains("free_bytes"));
372                        assert!(disk_keys.contains("available_bytes"));
373
374                        // Should have percentage metrics
375                        assert!(disk_keys.contains("usage_percent"));
376                        assert!(disk_keys.contains("available_percent"));
377                        assert!(disk_keys.contains("disk_pressure"));
378
379                        // Should have threshold checks
380                        assert!(disk_keys.contains("usage_warning"));
381                        assert!(disk_keys.contains("usage_critical"));
382
383                        // Should have I/O info
384                        assert!(disk_keys.contains("io_supported"));
385                    } else {
386                        panic!("Disk entry should be a mapping");
387                    }
388                }
389            } else {
390                panic!("disks should be a sequence");
391            }
392        } else {
393            panic!("Expected mapping value");
394        }
395    }
