1//! Command facts collector
2//!
3//! Executes custom commands and collects their output as facts.
4//!
5//! # Examples
6//!
7//! ## Basic command output collection
8//!
9//! **YAML Format:**
10//! ```yaml
11//! type: command
12//! name: uptime
13//! command: uptime -p
14//! format: text
15//! labels:
16//!   category: system
17//! ```
18//!
19//! **JSON Format:**
20//! ```json
21//! {
22//!   "type": "command",
23//!   "name": "uptime",
24//!   "command": "uptime -p",
25//!   "format": "text",
26//!   "labels": {
27//!     "category": "system"
28//!   }
29//! }
30//! ```
31//!
32//! **TOML Format:**
33//! ```toml
34//! [[collectors]]
35//! type = "command"
36//! name = "uptime"
37//! command = "uptime -p"
38//! format = "text"
39//!
40//! [collectors.labels]
41//! category = "system"
42//! ```
43//!
44//! ## JSON command output parsing
45//!
46//! **YAML Format:**
47//! ```yaml
48//! type: command
49//! name: docker_stats
50//! command: docker stats --no-stream --format json
51//! format: json
52//! cwd: /tmp
53//! env:
54//!   DOCKER_HOST: unix:///var/run/docker.sock
55//! ```
56//!
57//! **JSON Format:**
58//! ```json
59//! {
60//!   "type": "command",
61//!   "name": "docker_stats",
62//!   "command": "docker stats --no-stream --format json",
63//!   "format": "json",
64//!   "cwd": "/tmp",
65//!   "env": {
66//!     "DOCKER_HOST": "unix:///var/run/docker.sock"
67//!   }
68//! }
69//! ```
70//!
71//! **TOML Format:**
72//! ```toml
73//! [[collectors]]
74//! type = "command"
75//! name = "docker_stats"
76//! command = "docker stats --no-stream --format json"
77//! format = "json"
78//! cwd = "/tmp"
79//!
80//! [collectors.env]
81//! DOCKER_HOST = "unix:///var/run/docker.sock"
82//! ```
83//!
84//! **Output:**
85//! ```yaml
86//! command: "docker stats --no-stream --format json"
87//! exit_code: 0
88//! output:
89//!   - container: "web_server"
90//!     cpu_percent: "5.2"
91//!     memory_usage: "128MiB / 1GiB"
92//!     net_io: "1.2kB / 3.4kB"
93//!   - container: "database"
94//!     cpu_percent: "2.1"
95//!     memory_usage: "256MiB / 2GiB"
96//!     net_io: "500B / 1.2kB"
97//! labels:
98//!   category: monitoring
99//! ```
100//!
101//! ## Key-value command output parsing
102//!
103//! **YAML Format:**
104//! ```yaml
105//! type: command
106//! name: system_info
107//! command: echo "hostname=$(hostname)\nos_version=$(cat /etc/os-release | grep PRETTY_NAME | cut -d'=' -f2 | tr -d '\"')\nuptime=$(uptime -p)"
108//! format: key_value
109//! labels:
110//!   category: system
111//! ```
112//!
113//! **JSON Format:**
114//! ```json
115//! {
116//!   "type": "command",
117//!   "name": "system_info",
118//!   "command": "echo \"hostname=$(hostname)\\nos_version=$(cat /etc/os-release | grep PRETTY_NAME | cut -d'=' -f2 | tr -d '\\\"')\\nuptime=$(uptime -p)\"",
119//!   "format": "key_value",
120//!   "labels": {
121//!     "category": "system"
122//!   }
123//! }
124//! ```
125//!
126//! **TOML Format:**
127//! ```toml
128//! [[collectors]]
129//! type = "command"
130//! name = "system_info"
131//! command = "echo \"hostname=$(hostname)\\nos_version=$(cat /etc/os-release | grep PRETTY_NAME | cut -d'=' -f2 | tr -d '\\\"')\\nuptime=$(uptime -p)\""
132//! format = "key_value"
133//!
134//! [collectors.labels]
135//! category = "system"
136//! ```
137//!
138//! **Output:**
139//! ```yaml
140//! command: "echo \"hostname=$(hostname)\\nos_version=$(cat /etc/os-release | grep PRETTY_NAME | cut -d'=' -f2 | tr -d '\\\"')\\nuptime=$(uptime -p)\""
141//! exit_code: 0
142//! output:
143//!   hostname: "web-server-01"
144//!   os_version: "Ubuntu 22.04.3 LTS"
145//!   uptime: "up 2 weeks, 3 days, 4 hours"
146//! labels:
147//!   category: system
148//! ```

154//! type: command
155//! name: disk_usage
156//! command: df -h /

158//! labels:
159//!   category: storage
160//! ```
161//!
162//! **JSON Format:**
163//! ```json
164//! {
165//!   "type": "command",
166//!   "name": "disk_usage",
167//!   "command": "df -h /",
168//!   "format": "text",
169//!   "labels": {
170//!     "category": "storage"
171//!   }
172//! }
173//! ```
174//!
175//! **TOML Format:**
176//! ```toml
177//! [[collectors]]
178//! type = "command"
179//! name = "disk_usage"
180//! command = "df -h /"
181//! format = "text"
182//!
183//! [collectors.labels]
184//! category = "storage"
185//! ```
186//!
187//! **Output:**
188//! ```yaml
189//! command: "df -h /"
190//! exit_code: 0
191//! stdout: |
192//!   Filesystem      Size  Used Avail Use% Mounted on
193//!   /dev/sda1        50G   15G   33G  31% /
194//! labels:
195//!   category: storage
196//! ```

198//! ## Command with environment variables and working directory
199//!
200//! **YAML Format:**
201//! ```yaml
202//! type: command
203//! name: custom_script
204//! command: ./check_service.sh
205//! format: json
206//! cwd: /opt/myapp
207//! env:
208//!   SERVICE_NAME: myapp
209//!   LOG_LEVEL: info
210//! labels:
211//!   category: application
212//! ```
213//!
214//! **JSON Format:**
215//! ```json
216//! {
217//!   "type": "command",
218//!   "name": "custom_script",
219//!   "command": "./check_service.sh",
220//!   "format": "json",
221//!   "cwd": "/opt/myapp",
222//!   "env": {
223//!     "SERVICE_NAME": "myapp",
224//!     "LOG_LEVEL": "info"
225//!   },
226//!   "labels": {
227//!     "category": "application"
228//!   }
229//! }
230//! ```
231//!
232//! **TOML Format:**
233//! ```toml
234//! [[collectors]]
235//! type = "command"
236//! name = "custom_script"
237//! command = "./check_service.sh"
238//! format = "json"
239//! cwd = "/opt/myapp"
240//!
241//! [collectors.env]
242//! SERVICE_NAME = "myapp"
243//! LOG_LEVEL = "info"
244//!
245//! [collectors.labels]
246//! category = "application"
247//! ```
248//!
249//! **Output:**
250//! ```yaml
251//! command: "./check_service.sh"
252//! exit_code: 0
253//! output:
254//!   service_status: "running"
255//!   uptime_seconds: 3600
256//!   version: "1.2.3"
257//!   health_checks:
258//!     - name: "database"
259//!       status: "ok"
260//!     - name: "cache"
261//!       status: "ok"
262//! labels:
263//!   category: application
264//! ```

266use crate::facts::{CommandCollector, CommandOutputFormat};
267use anyhow::Result;
268use serde_yaml::Value;
269use std::collections::HashMap;
270use std::process::Command;
271
272/// Execute command facts collection
273pub fn collect_command_facts(collector: &CommandCollector) -> Result<Value> {
274    let mut facts = HashMap::new();
275
276    // Parse the command string (simple parsing for now)
277    let parts: Vec<&str> = collector.command.split_whitespace().collect();
278    if parts.is_empty() {
279        return Err(anyhow::anyhow!("Empty command"));
280    }
281
282    let program = parts[0];
283    let args = &parts[1..];
284
285    // Build the command
286    let mut cmd = Command::new(program);
287    cmd.args(args);
288
289    // Set working directory if specified
290    if let Some(cwd) = &collector.cwd {
291        cmd.current_dir(cwd);
292    }
293
294    // Set environment variables
295    for (key, value) in &collector.env {
296        cmd.env(key, value);
297    }
298
299    // Execute the command
300    let output = cmd.output()?;
301
302    // Store basic command information
303    facts.insert(
304        "command".to_string(),
305        Value::String(collector.command.clone()),
306    );
307    facts.insert(
308        "exit_code".to_string(),
309        Value::Number(output.status.code().unwrap_or(-1).into()),
310    );
311
312    // Process stdout based on format
313    let stdout_str = String::from_utf8_lossy(&output.stdout);
314    match collector.format {
315        CommandOutputFormat::Text => {
316            facts.insert("stdout".to_string(), Value::String(stdout_str.to_string()));
317        }
318        CommandOutputFormat::Json => {
319            // Try to parse as JSON
320            match serde_json::from_str::<serde_json::Value>(&stdout_str) {
321                Ok(json_value) => {
322                    facts.insert("output".to_string(), serde_yaml::to_value(&json_value)?);
323                }
324                Err(_) => {
325                    // Fallback to text if JSON parsing fails
326                    facts.insert("stdout".to_string(), Value::String(stdout_str.to_string()));
327                    facts.insert(
328                        "parse_error".to_string(),
329                        Value::String("Failed to parse as JSON".to_string()),
330                    );
331                }
332            }
333        }
334        CommandOutputFormat::KeyValue => {
335            // Parse key=value format
336            let mut parsed = HashMap::new();
337            for line in stdout_str.lines() {
338                if let Some((key, value)) = line.split_once('=') {
339                    parsed.insert(
340                        key.trim().to_string(),
341                        Value::String(value.trim().to_string()),
342                    );
343                }
344            }
345            facts.insert(
346                "output".to_string(),
347                Value::Mapping(
348                    parsed
349                        .into_iter()
350                        .map(|(k, v)| (Value::String(k), v))
351                        .collect(),
352                ),
353            );
354        }
355    }
356
357    // Process stderr
358    let stderr_str = String::from_utf8_lossy(&output.stderr);
359    if !stderr_str.trim().is_empty() {
360        facts.insert("stderr".to_string(), Value::String(stderr_str.to_string()));
361    }
362
363    // Add configured labels
364    if !collector.labels.is_empty() {
365        let mut labels: HashMap<String, Value> = HashMap::new();
366        for (key, value) in &collector.labels {
367            labels.insert(key.clone(), Value::String(value.clone()));
368        }
369        facts.insert(
370            "labels".to_string(),
371            Value::Mapping(
372                labels
373                    .into_iter()
374                    .map(|(k, v)| (Value::String(k), v))
375                    .collect(),
376            ),
377        );
378    }
379
380    // Add base labels if any
381    if !collector.base.labels.is_empty() {
382        let mut base_labels: HashMap<String, Value> = HashMap::new();
383        for (key, value) in &collector.base.labels {
384            base_labels.insert(key.clone(), Value::String(value.clone()));
385        }
386        facts.insert(
387            "base_labels".to_string(),
388            Value::Mapping(
389                base_labels
390                    .into_iter()
391                    .map(|(k, v)| (Value::String(k), v))
392                    .collect(),
393            ),
394        );
395    }
396
397    Ok(Value::Mapping(
398        facts
399            .into_iter()
400            .map(|(k, v)| (Value::String(k), v))
401            .collect(),
402    ))
403}

411    #[test]
412    fn test_collect_command_facts_text_format() {
413        let collector = CommandCollector {
414            base: BaseCollector {
415                name: "test_command".to_string(),
416                enabled: true,
417                poll_interval: 60,
418                labels: HashMap::new(),
419            },
420            command: "echo 'Hello World'".to_string(),
421            format: CommandOutputFormat::Text,
422            cwd: None,
423            env: HashMap::new(),
424            labels: HashMap::new(),
425        };
426
427        let result = collect_command_facts(&collector);
428        assert!(result.is_ok());
429
430        let value = result.unwrap();
431        if let Value::Mapping(map) = value {
432            let keys: std::collections::HashSet<_> = map
433                .keys()
434                .filter_map(|k| {
435                    if let Value::String(s) = k {
436                        Some(s.as_str())
437                    } else {
438                        None
439                    }
440                })
441                .collect();
442
443            assert!(keys.contains("command"));
444            assert!(keys.contains("exit_code"));
445            assert!(keys.contains("stdout"));
446        } else {
447            panic!("Expected mapping value");
448        }
449    }

631        };

654            );
655            assert!(keys.contains("command"));
656        } else {
