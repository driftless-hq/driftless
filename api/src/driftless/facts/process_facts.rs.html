1//! Process facts collector
2//!
3//! Collects process information and resource usage statistics.
4//!
5//! # Examples
6//!
7//! ## Basic process metrics collection
8//!
9//! **YAML Format:**
10//! ```yaml
11//! type: process
12//! name: process
13//! patterns: ["nginx", "apache", "sshd"]
14//! collect:
15//!   count: true
16//!   cpu: true
17//!   memory: true
18//!   status: true
19//! ```
20//!
21//! **JSON Format:**
22//! ```json
23//! {
24//!   "type": "process",
25//!   "name": "process",
26//!   "patterns": ["nginx", "apache", "sshd"],
27//!   "collect": {
28//!     "count": true,
29//!     "cpu": true,
30//!     "memory": true,
31//!     "status": true
32//!   }
33//! }
34//! ```
35//!
36//! **TOML Format:**
37//! ```toml
38//! [[collectors]]
39//! type = "process"
40//! name = "process"
41//! patterns = ["nginx", "apache", "sshd"]
42//!
43//! [collectors.collect]
44//! count = true
45//! cpu = true
46//! memory = true
47//! status = true
48//! ```
49//!
50//! **Output:**
51//! ```yaml
52//! total_processes: 150
53//! matched_processes: 3
54//! processes:
55//!   - pid: 1234
56//!     name: "nginx"
57//!     cpu_percent: 5
58//!     memory_bytes: 104857600
59//!     memory_mb: 100
60//!     memory_gb: 0
61//!     status: "running"
62//!     command: "/usr/sbin/nginx"
63//!     parent_pid: 1
64//!   - pid: 1235
65//!     name: "nginx"
66//!     cpu_percent: 3
67//!     memory_bytes: 52428800
68//!     memory_mb: 50
69//!     memory_gb: 0
70//!     status: "running"
71//!     command: "/usr/sbin/nginx"
72//!     parent_pid: 1234
73//!   - pid: 5678
74//!     name: "apache2"
75//!     cpu_percent: 2
76//!     memory_bytes: 209715200
77//!     memory_mb: 200
78//!     memory_gb: 0
79//!     status: "sleeping"
80//!     command: "/usr/sbin/apache2"
81//!     parent_pid: 1
82//! labels:
83//!   process_type: web_servers
84//! ```

86use crate::facts::ProcessCollector;
87use anyhow::Result;
88use serde_yaml::Value;
89use std::collections::HashMap;
90use sysinfo::System;

92/// Execute process facts collection
93pub fn collect_process_facts(collector: &ProcessCollector) -> Result<Value> {
94    let mut system = System::new();
95    system.refresh_all();
96
97    let mut facts = HashMap::new();
98    let mut processes_info = Vec::new();
99    let total_processes = system.processes().len();
100
101    // Compile regex patterns for filtering
102    let patterns: Vec<regex::Regex> = collector
103        .patterns
104        .iter()
105        .filter_map(|pattern| regex::Regex::new(pattern).ok())
106        .collect();
107
108    let mut matched_processes = 0;
109
110    // Iterate over all processes
111    for (pid, process) in system.processes() {
112        let process_name = process.name().to_string();
113
114        // Filter by patterns if specified
115        let matches_pattern = if !patterns.is_empty() {
116            patterns
117                .iter()
118                .any(|pattern| pattern.is_match(&process_name))
119        } else {
120            true // No patterns means include all processes
121        };
122
123        if !matches_pattern {
124            continue;
125        }
126
127        matched_processes += 1;
128
129        let mut process_info = HashMap::new();
130
131        // Basic process info
132        process_info.insert("pid".to_string(), Value::Number(pid.as_u32().into()));
133        process_info.insert("name".to_string(), Value::String(process_name.clone()));
134
135        // Collect CPU usage
136        if collector.collect.cpu {
137            let cpu_usage = process.cpu_usage();
138            process_info.insert(
139                "cpu_percent".to_string(),
140                Value::Number((cpu_usage as i64).into()),
141            );
142        }
143
144        // Collect memory usage
145        if collector.collect.memory {
146            let memory_usage = process.memory();
147            process_info.insert(
148                "memory_bytes".to_string(),
149                Value::Number(memory_usage.into()),
150            );
151            process_info.insert(
152                "memory_mb".to_string(),
153                Value::Number((memory_usage / 1024 / 1024).into()),
154            );
155            process_info.insert(
156                "memory_gb".to_string(),
157                Value::Number((memory_usage / 1024 / 1024 / 1024).into()),
158            );
159        }
160
161        // Collect process status
162        if collector.collect.status {
163            let status = match process.status() {
164                sysinfo::ProcessStatus::Run => "running",
165                sysinfo::ProcessStatus::Sleep => "sleeping",
166                sysinfo::ProcessStatus::Stop => "stopped",
167                sysinfo::ProcessStatus::Zombie => "zombie",
168                sysinfo::ProcessStatus::Tracing => "tracing",
169                sysinfo::ProcessStatus::Dead => "dead",
170                sysinfo::ProcessStatus::Wakekill => "wakekill",
171                sysinfo::ProcessStatus::Waking => "waking",
172                sysinfo::ProcessStatus::Parked => "parked",
173                sysinfo::ProcessStatus::Idle => "idle",
174                _ => "unknown",
175            };
176            process_info.insert("status".to_string(), Value::String(status.to_string()));
177
178            // Add additional process info
179            if let Some(cmd) = process.cmd().first() {
180                process_info.insert("command".to_string(), Value::String(cmd.clone()));
181            }
182            process_info.insert(
183                "parent_pid".to_string(),
184                Value::Number(process.parent().map(|p| p.as_u32()).unwrap_or(0).into()),
185            );
186        }
187
188        processes_info.push(Value::Mapping(
189            process_info
190                .into_iter()
191                .map(|(k, v)| (Value::String(k), v))
192                .collect(),
193        ));
194    }
195
196    // Collect process count
197    if collector.collect.count {
198        facts.insert(
199            "total_processes".to_string(),
200            Value::Number(total_processes.into()),
201        );
202        facts.insert(
203            "matched_processes".to_string(),
204            Value::Number(matched_processes.into()),
205        );
206    }
207
208    facts.insert("processes".to_string(), Value::Sequence(processes_info));
209
210    // Add base labels if any
211    if !collector.base.labels.is_empty() {
212        let mut labels = HashMap::new();
213        for (key, value) in &collector.base.labels {
214            labels.insert(key.clone(), Value::String(value.clone()));
215        }
216        facts.insert(
217            "labels".to_string(),
218            Value::Mapping(
219                labels
220                    .into_iter()
221                    .map(|(k, v)| (Value::String(k), v))
222                    .collect(),
223            ),
224        );
225    }
226
227    Ok(Value::Mapping(
228        facts
229            .into_iter()
230            .map(|(k, v)| (Value::String(k), v))
231            .collect(),
232    ))
233}

241    #[test]
242    fn test_collect_process_facts_basic() {
243        let collector = ProcessCollector {
244            base: BaseCollector {
245                name: "process".to_string(),
246                enabled: true,
247                poll_interval: 60,
248                labels: HashMap::new(),
249            },
250            patterns: vec!["nginx".to_string(), "apache".to_string()],
251            collect: ProcessCollectOptions {
252                count: true,
253                cpu: true,
254                memory: true,
255                status: true,
256            },
257        };
258
259        let result = collect_process_facts(&collector);
260        assert!(result.is_ok());
261
262        let value = result.unwrap();
263        if let Value::Mapping(map) = value {
264            let keys: std::collections::HashSet<_> = map
265                .keys()
266                .filter_map(|k| {
267                    if let Value::String(s) = k {
268                        Some(s.as_str())
269                    } else {
270                        None
271                    }
272                })
273                .collect();
274
275            // Should have processes array
276            assert!(keys.contains("processes"));
277            assert!(keys.contains("total_processes"));
278            assert!(keys.contains("matched_processes"));
279
280            // Check processes array
281            if let Some(Value::Sequence(processes)) =
282                map.get(&Value::String("processes".to_string()))
283            {
284                // If there are processes, validate structure
285                for process in processes {
286                    if let Value::Mapping(process_map) = process {
287                        let process_keys: std::collections::HashSet<_> = process_map
288                            .keys()
289                            .filter_map(|k| {
290                                if let Value::String(s) = k {
291                                    Some(s.as_str())
292                                } else {
293                                    None
294                                }
295                            })
296                            .collect();
297
298                        // Should have basic fields
299                        assert!(process_keys.contains("pid"));
300                        assert!(process_keys.contains("name"));
301
302                        // Should have CPU metrics
303                        assert!(process_keys.contains("cpu_percent"));
304
305                        // Should have memory metrics
306                        assert!(process_keys.contains("memory_bytes"));
307                        assert!(process_keys.contains("memory_mb"));
308                        assert!(process_keys.contains("memory_gb"));
309
310                        // Should have status
311                        assert!(process_keys.contains("status"));
312                        assert!(process_keys.contains("command"));
313                        assert!(process_keys.contains("parent_pid"));
314                    } else {
315                        panic!("Process entry should be a mapping");
316                    }
317                }
318            } else {
319                panic!("processes should be a sequence");
320            }
321        } else {
322            panic!("Expected mapping value");
323        }
324    }
