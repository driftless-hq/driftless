1//! Package task executor
2//!
3//! Handles package management operations: install, remove, update packages
4//! using the appropriate package manager for the system.
5//!
6//! # Examples
7//!
8//! ## Install a package
9//!
10//! This example installs nginx using the system's default package manager.
11//!
12//! **YAML Format:**
13//! ```yaml
14//! - type: package
15//!   description: "Install nginx web server"
16//!   name: nginx
17//!   state: present
18//! ```
19//!
20//! **JSON Format:**
21//! ```json
22//! {
23//!   "type": "package",
24//!   "description": "Install nginx web server",
25//!   "name": "nginx",
26//!   "state": "present"
27//! }
28//! ```
29//!
30//! **TOML Format:**
31//! ```toml
32//! [[tasks]]
33//! type = "package"
34//! description = "Install nginx web server"
35//! name = "nginx"
36//! state = "present"
37//! ```
38//!
39//! ## Install with specific package manager
40//!
41//! This example forces the use of apt even on systems that might have multiple package managers.
42//!
43//! **YAML Format:**
44//! ```yaml
45//! - type: package
46//!   description: "Install curl using apt"
47//!   name: curl
48//!   state: present
49//!   manager: apt
50//! ```
51//!
52//! **JSON Format:**
53//! ```json
54//! {
55//!   "type": "package",
56//!   "description": "Install curl using apt",
57//!   "name": "curl",
58//!   "state": "present",
59//!   "manager": "apt"
60//! }
61//! ```
62//!
63//! **TOML Format:**
64//! ```toml
65//! [[tasks]]
66//! type = "package"
67//! description = "Install curl using apt"
68//! name = "curl"
69//! state = "present"
70//! manager = "apt"
71//! ```

185/// Package state enumeration
186#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
187#[serde(rename_all = "lowercase")]
188pub enum PackageState {
189    /// Ensure package is installed
190    Present,
191    /// Ensure package is not installed
192    Absent,
193    /// Ensure package is latest version
194    Latest,
195}
196
197/// Package management task
198///
199/// # Registered Outputs
200/// - `changed` (bool): Whether any packages were installed or removed
201/// - `packages` (`Vec<String>`): List of packages affected
202#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
203pub struct PackageTask {
204    /// Optional description of what this task does
205    ///
206    /// Human-readable description of the task's purpose. Used for documentation
207    /// and can be displayed in logs or reports.
208    #[serde(skip_serializing_if = "Option::is_none")]
209    pub description: Option<String>,
210
211    /// Package name
212    pub name: String,
213    /// Package state
214    pub state: PackageState,
215    /// Package manager to use (auto-detect if not specified)
216    #[serde(skip_serializing_if = "Option::is_none")]
217    pub manager: Option<String>,
218}

223/// Execute a package task
224pub async fn execute_package_task(task: &PackageTask, dry_run: bool) -> Result<serde_yaml::Value> {
225    let manager = detect_package_manager()
226        .or_else(|| task.manager.as_ref().cloned())
227        .ok_or_else(|| anyhow::anyhow!("Could not detect package manager"))?;
228
229    let changed = match task.state {
230        PackageState::Present => ensure_package_present(&task.name, &manager, dry_run).await?,
231        PackageState::Absent => ensure_package_absent(&task.name, &manager, dry_run).await?,
232        PackageState::Latest => ensure_package_latest(&task.name, &manager, dry_run).await?,
233    };
234
235    let mut result = serde_yaml::Mapping::new();
236    result.insert(
237        serde_yaml::Value::from("changed"),
238        serde_yaml::Value::from(changed),
239    );
240
241    let packages = vec![serde_yaml::Value::from(task.name.clone())];
242    result.insert(
243        serde_yaml::Value::from("packages"),
244        serde_yaml::Value::from(packages),
245    );
246
247    Ok(serde_yaml::Value::Mapping(result))
248}

250/// Detect the package manager available on the system
251fn detect_package_manager() -> Option<String> {
252    let managers = vec![
253        ("apt-get", "apt"),
254        ("yum", "yum"),
255        ("dnf", "dnf"),
256        ("pacman", "pacman"),
257        ("zypper", "zypper"),
258        ("brew", "brew"),
259    ];
260
261    for (cmd, name) in managers {
262        if Command::new("which")
263            .arg(cmd)
264            .output()
265            .map(|o| o.status.success())
266            .unwrap_or(false)
267        {
268            return Some(name.to_string());
269        }
270    }
271
272    None
273}

275/// Ensure a package is present
276async fn ensure_package_present(package: &str, manager: &str, dry_run: bool) -> Result<bool> {
277    if is_package_installed(package, manager)? {
278        println!("Package {} is already installed", package);
279        return Ok(false);
280    }
281
282    let install_cmd = get_install_command(package, manager);
283
284    if dry_run {
285        println!("Would run: {}", install_cmd.join(" "));
286    } else {
287        run_command(&install_cmd)
288            .with_context(|| format!("Failed to install package {}", package))?;
289        println!("Installed package: {}", package);
290    }
291
292    Ok(true)
293}

295/// Ensure a package is not installed
296async fn ensure_package_absent(package: &str, manager: &str, dry_run: bool) -> Result<bool> {
297    if !is_package_installed(package, manager)? {
298        println!("Package {} is not installed", package);
299        return Ok(false);
300    }
301
302    let remove_cmd = get_remove_command(package, manager);
303
304    if dry_run {
305        println!("Would run: {}", remove_cmd.join(" "));
306    } else {
307        run_command(&remove_cmd)
308            .with_context(|| format!("Failed to remove package {}", package))?;
309        println!("Removed package: {}", package);
310    }
311
312    Ok(true)
313}

315/// Ensure a package is at the latest version
316async fn ensure_package_latest(package: &str, manager: &str, dry_run: bool) -> Result<bool> {
317    let upgrade_cmd = get_upgrade_command(package, manager);
318
319    if dry_run {
320        println!("Would run: {}", upgrade_cmd.join(" "));
321        Ok(true)
322    } else {
323        run_command(&upgrade_cmd)
324            .with_context(|| format!("Failed to upgrade package {}", package))?;
325        println!("Upgraded package: {}", package);
326        Ok(true)
327    }
328}

330/// Check if a package is installed
331fn is_package_installed(package: &str, manager: &str) -> Result<bool> {
332    let check_cmd = get_check_command(package, manager);
333
334    let output = Command::new(&check_cmd[0])
335        .args(&check_cmd[1..])
336        .output()
337        .with_context(|| format!("Failed to check if package {} is installed", package))?;
338
339    Ok(output.status.success())
340}

478/// Get the check command for a package manager
479fn get_check_command(package: &str, manager: &str) -> Vec<String> {
480    match manager {
481        "apt" => vec!["dpkg".to_string(), "-l".to_string(), package.to_string()],
482        "yum" | "dnf" => vec!["rpm".to_string(), "-q".to_string(), package.to_string()],
483        "pacman" => vec!["pacman".to_string(), "-Q".to_string(), package.to_string()],
484        "zypper" => vec!["rpm".to_string(), "-q".to_string(), package.to_string()],
485        "brew" => vec!["brew".to_string(), "list".to_string(), package.to_string()],
486        _ => vec![
487            "echo".to_string(),
488            format!("Unsupported package manager: {}", manager),
489        ],
490    }
491}

493/// Run a command and return the result
494fn run_command(cmd: &[String]) -> Result<()> {
495    if cmd.is_empty() {
496        return Ok(());
497    }
498
499    let status = Command::new(&cmd[0])
500        .args(&cmd[1..])
501        .status()
502        .with_context(|| format!("Failed to execute command: {}", cmd.join(" ")))?;
503
504    if status.success() {
505        Ok(())
506    } else {
507        Err(anyhow::anyhow!(
508            "Command failed with exit code: {}",
509            status.code().unwrap_or(-1)
510        ))
511    }
512}
