7//! ## Set system hostname
8//!
9//! This example sets the system hostname to "web-server-01".
10//!
11//! **YAML Format:**
12//! ```yaml
13//! - type: hostname
14//!   description: "Set system hostname"
15//!   name: web-server-01
16//!   persist: true
17//! ```
18//!
19//! **JSON Format:**
20//! ```json
21//! {
22//!   "type": "hostname",
23//!   "description": "Set system hostname",
24//!   "name": "web-server-01",
25//!   "persist": true
26//! }
27//! ```
28//!
29//! **TOML Format:**
30//! ```toml
31//! [[tasks]]
32//! type = "hostname"
33//! description = "Set system hostname"
34//! name = "web-server-01"
35//! persist = true
36//! ```
37//!
38//! ## Set hostname temporarily

100/// System hostname management task
101#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
102pub struct HostnameTask {
103    /// Optional description of what this task does
104    ///
105    /// Human-readable description of the task's purpose. Used for documentation
106    /// and can be displayed in logs or reports.
107    #[serde(skip_serializing_if = "Option::is_none")]
108    pub description: Option<String>,
109
110    /// Desired hostname
111    pub name: String,
112    /// Whether to persist hostname to /etc/hostname
113    #[serde(default = "crate::apply::default_true")]
114    pub persist: bool,
115}
116
117use anyhow::{Context, Result};
118use std::fs;
119
120/// Execute a hostname task
121pub async fn execute_hostname_task(task: &HostnameTask, dry_run: bool) -> Result<()> {
122    // Get current hostname
123    let current_hostname = get_current_hostname()?;
124
125    if current_hostname == task.name {
126        println!("Hostname already set to: {}", task.name);
127        return Ok(());
128    }
129
130    println!(
131        "Changing hostname from '{}' to '{}'",
132        current_hostname, task.name
133    );
134
135    if dry_run {
136        println!("Would set hostname to: {}", task.name);
137        if task.persist {
138            println!("  and persist to /etc/hostname");
139        }
140    } else {
141        set_hostname(&task.name)?;
142        println!("Set hostname to: {}", task.name);
143
144        if task.persist {
145            persist_hostname(&task.name)?;
146            println!("Persisted hostname to /etc/hostname");
147        }
148    }
149
150    Ok(())
151}
152
153/// Get the current system hostname
154fn get_current_hostname() -> Result<String> {
155    let output = std::process::Command::new("hostname")
156        .output()
157        .with_context(|| "Failed to get current hostname")?;
158
159    Ok(String::from_utf8_lossy(&output.stdout).trim().to_string())
160}
161
162/// Set the current hostname
163fn set_hostname(hostname: &str) -> Result<()> {
164    std::process::Command::new("hostname")
165        .arg(hostname)
166        .status()
167        .with_context(|| format!("Failed to set hostname to {}", hostname))?
168        .success()
169        .then_some(())
170        .ok_or_else(|| anyhow::anyhow!("hostname command failed"))?;
171
172    Ok(())
173}
174
175/// Persist hostname to /etc/hostname
176fn persist_hostname(hostname: &str) -> Result<()> {
177    fs::write("/etc/hostname", format!("{}\n", hostname))
178        .with_context(|| "Failed to write hostname to /etc/hostname".to_string())?;
179
180    Ok(())
181}

189        let task = HostnameTask {
190            description: None,
