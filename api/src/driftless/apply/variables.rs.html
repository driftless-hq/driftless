10/// Variable storage for task execution context
11#[derive(Debug, Clone, Default)]
12pub struct VariableContext {
13    variables: HashMap<String, serde_yaml::Value>,
14    /// Built-in facts (system information)
15    facts: HashMap<String, serde_yaml::Value>,
16}

44        );

46        // Load environment variables into driftless_env

50    /// Load environment variables into driftless_env fact
51    fn load_environment_variables(&mut self) {
52        let mut env_vars = Mapping::new();
53        for (key, value) in std::env::vars() {
54            env_vars.insert(
55                serde_yaml::Value::String(key),
56                serde_yaml::Value::String(value),
57            );
58        }
59        self.facts
60            .insert("env".to_string(), serde_yaml::Value::Mapping(env_vars));
61    }

63    /// Load variables from env file
64    ///
65    /// Supports .env format: KEY=value
66    /// Variables are added to the 'env' fact for template access
67    pub fn load_env_file(
68        &mut self,
69        path: &std::path::Path,
70    ) -> Result<(), Box<dyn std::error::Error>> {
71        if !path.exists() {
72            return Ok(()); // Silently ignore missing env files
73        }
74
75        let content = std::fs::read_to_string(path)?;
76
77        // Ensure we have an env mapping
78        if !self.facts.contains_key("env") {
79            self.facts.insert(
80                "env".to_string(),
81                serde_yaml::Value::Mapping(Mapping::new()),
82            );
83        }
84
85        if let Some(serde_yaml::Value::Mapping(ref mut env_map)) = self.facts.get_mut("env") {
86            for line in content.lines() {
87                let line = line.trim();
88                if line.is_empty() || line.starts_with('#') {
89                    continue;
90                }
91
92                if let Some(eq_pos) = line.find('=') {
93                    let key = line[..eq_pos].trim();
94                    let value = line[eq_pos + 1..].trim();
95
96                    // Remove surrounding quotes if present
97                    let value = if (value.starts_with('"') && value.ends_with('"'))
98                        || (value.starts_with('\'') && value.ends_with('\''))
99                    {
100                        &value[1..value.len() - 1]
101                    } else {
102                        value
103                    };
104
105                    env_map.insert(
106                        serde_yaml::Value::String(key.to_string()),
107                        serde_yaml::Value::String(value.to_string()),
108                    );
109                }
110            }
111        }
112
113        Ok(())
114    }

131    /// Create a minijinja context with all variables and facts
132    fn create_jinja_context(&self) -> minijinja::Value {
133        let mut context = serde_json::Map::new();
134
135        // Add variables
136        for (key, value) in &self.variables {
137            context.insert(key.clone(), self.yaml_to_json(value));
138        }
139
140        // Add facts
141        for (key, value) in &self.facts {
142            context.insert(key.clone(), self.yaml_to_json(value));
143        }
144
145        minijinja::Value::from_serialize(&context)
146    }

148    /// Convert serde_yaml::Value to serde_json::Value for minijinja
149    fn yaml_to_json(&self, value: &serde_yaml::Value) -> serde_json::Value {
150        match value {
151            serde_yaml::Value::Null => serde_json::Value::Null,
152            serde_yaml::Value::Bool(b) => serde_json::Value::Bool(*b),
153            serde_yaml::Value::Number(n) => {
154                if let Some(i) = n.as_i64() {
155                    serde_json::Value::Number(serde_json::Number::from(i))
156                } else if let Some(f) = n.as_f64() {
157                    serde_json::Number::from_f64(f)
158                        .map(serde_json::Value::Number)
159                        .unwrap_or(serde_json::Value::Null)
160                } else {
161                    serde_json::Value::Null
162                }
163            }
164            serde_yaml::Value::String(s) => serde_json::Value::String(s.clone()),
165            serde_yaml::Value::Sequence(seq) => {
166                let json_seq: Vec<serde_json::Value> =
167                    seq.iter().map(|v| self.yaml_to_json(v)).collect();
168                serde_json::Value::Array(json_seq)
169            }
170            serde_yaml::Value::Mapping(map) => {
171                let mut json_map = serde_json::Map::new();
172                for (k, v) in map {
173                    if let serde_yaml::Value::String(key) = k {
174                        json_map.insert(key.clone(), self.yaml_to_json(v));
175                    }
176                }
177                serde_json::Value::Object(json_map)
178            }
179            serde_yaml::Value::Tagged(_) => serde_json::Value::Null, // Not supported
180        }
181    }

183    /// Render a template string with variable substitution

191    /// Evaluate a boolean expression
192    ///
193    /// Supports complex Driftless expressions with variables, comparisons, and logical operators
194    pub fn evaluate_condition(&self, condition: &str) -> bool {
195        let trimmed = condition.trim();
196
197        // Handle simple boolean literals
198        match trimmed.to_lowercase().as_str() {
199            "true" | "yes" | "1" => return true,
200            "false" | "no" | "0" => return false,
201            _ => {}
202        }
203
204        // Render template expressions first
205        let rendered = self.render_template(trimmed);
206
207        // Parse the rendered expression
208        self.evaluate_boolean_expression(&rendered)
209    }

211    /// Evaluate a boolean expression after template rendering
212    fn evaluate_boolean_expression(&self, expr: &str) -> bool {
213        let expr = expr.trim();
214
215        // Handle logical NOT
216        if expr.starts_with("not ") || expr.starts_with("!") {
217            let rest = expr
218                .strip_prefix("not ")
219                .or_else(|| expr.strip_prefix('!'))
220                .unwrap_or(expr);
221            return !self.evaluate_boolean_expression(rest);
222        }
223
224        // Handle logical AND
225        if let Some(and_pos) = expr.find(" and ") {
226            let left = &expr[..and_pos];
227            let right = &expr[and_pos + 5..];
228            return self.evaluate_boolean_expression(left)
229                && self.evaluate_boolean_expression(right);
230        }
231
232        // Handle logical OR
233        if let Some(or_pos) = expr.find(" or ") {
234            let left = &expr[..or_pos];
235            let right = &expr[or_pos + 4..];
236            return self.evaluate_boolean_expression(left)
237                || self.evaluate_boolean_expression(right);
238        }
239
240        // Handle comparisons
241        if let Some(op_pos) = expr.find(" == ") {
242            let left = expr[..op_pos].trim();
243            let right = expr[op_pos + 4..].trim();
244            return self.compare_values(left, right, "==");
245        }
246        if let Some(op_pos) = expr.find(" != ") {
247            let left = expr[..op_pos].trim();
248            let right = expr[op_pos + 4..].trim();
249            return self.compare_values(left, right, "!=");
250        }
251        if let Some(op_pos) = expr.find(" < ") {
252            let left = expr[..op_pos].trim();
253            let right = expr[op_pos + 3..].trim();
254            return self.compare_values(left, right, "<");
255        }
256        if let Some(op_pos) = expr.find(" > ") {
257            let left = expr[..op_pos].trim();
258            let right = expr[op_pos + 3..].trim();
259            return self.compare_values(left, right, ">");
260        }
261        if let Some(op_pos) = expr.find(" <= ") {
262            let left = expr[..op_pos].trim();
263            let right = expr[op_pos + 4..].trim();
264            return self.compare_values(left, right, "<=");
265        }
266        if let Some(op_pos) = expr.find(" >= ") {
267            let left = expr[..op_pos].trim();
268            let right = expr[op_pos + 4..].trim();
269            return self.compare_values(left, right, ">=");
270        }
271
272        // Handle "is defined" checks
273        if let Some(var_name) = expr.strip_suffix(" is defined") {
274            return self.contains(var_name.trim());
275        }
276        if let Some(var_name) = expr.strip_suffix(" is not defined") {
277            return !self.contains(var_name.trim());
278        }
279
280        // Handle "in" operator
281        if let Some(in_pos) = expr.find(" in ") {
282            let item = expr[..in_pos].trim();
283            let container = expr[in_pos + 4..].trim();
284            return self.check_membership(item, container);
285        }
286
287        // Try to evaluate as a simple value
288        match expr.to_lowercase().as_str() {
289            "true" | "yes" => true,
290            "false" | "no" => false,
291            _ => {
292                // Check if it's a variable that evaluates to a boolean
293                // Since expr is already rendered, check if it's a truthy string
294                match expr.to_lowercase().as_str() {
295                    "true" | "yes" | "1" => true,
296                    "false" | "no" | "0" => false,
297                    "" => false,
298                    _ => !expr.is_empty(), // Non-empty strings are truthy
299                }
300            }
301        }
302    }

304    /// Compare two values
305    fn compare_values(&self, left: &str, right: &str, op: &str) -> bool {
306        // Try numeric comparison first
307        if let (Ok(left_num), Ok(right_num)) = (left.parse::<f64>(), right.parse::<f64>()) {
308            return match op {
309                "==" => left_num == right_num,
310                "!=" => left_num != right_num,
311                "<" => left_num < right_num,
312                ">" => left_num > right_num,
313                "<=" => left_num <= right_num,
314                ">=" => left_num >= right_num,
315                _ => false,
316            };
317        }
318
319        // String comparison
320        match op {
321            "==" => left == right,
322            "!=" => left != right,
323            "<" => left < right,
324            ">" => left > right,
325            "<=" => left <= right,
326            ">=" => left >= right,
327            _ => false,
328        }
329    }

331    /// Check if item is in container
332    fn check_membership(&self, item: &str, container_expr: &str) -> bool {
333        let container = container_expr.trim();
334
335        // Handle YAML sequence syntax like ["a", "b"]
336        if container.starts_with('[') && container.ends_with(']') {
337            let items_str = &container[1..container.len() - 1];
338            let items: Vec<&str> = items_str
339                .split(',')
340                .map(|s| s.trim().trim_matches('"').trim_matches('\''))
341                .collect();
342            return items.contains(&item);
343        }
344
345        // Check if container is a variable that holds a sequence
346        if let Some(serde_yaml::Value::Sequence(seq)) = self.get(container) {
347            return seq.iter().any(|v| match v {
348                serde_yaml::Value::String(s) => s == item,
349                serde_yaml::Value::Number(n) => n.to_string() == item,
350                _ => false,
351            });
352        }
353
354        false
355    }

381    #[test]
382    fn test_template_rendering() {
383        let mut ctx = VariableContext::new();
384        ctx.set(
385            "user".to_string(),
386            serde_yaml::Value::String("bob".to_string()),
387        );
388        ctx.set("count".to_string(), serde_yaml::Value::Number(42.into()));
389        ctx.set(
390            "path".to_string(),
391            serde_yaml::Value::String("/home/user/file.txt".to_string()),
392        );
393
394        // Basic variable substitution
395        assert_eq!(ctx.render_template("Hello {{ user }}!"), "Hello bob!");
396        assert_eq!(ctx.render_template("Count: {{ count }}"), "Count: 42");
397
398        // Filters
399        assert_eq!(ctx.render_template("{{ user | upper }}"), "BOB");
400        assert_eq!(ctx.render_template("{{ path | basename }}"), "file.txt");
401        assert_eq!(ctx.render_template("{{ path | dirname }}"), "/home/user");
402        assert_eq!(ctx.render_template("{{ user | capitalize }}"), "Bob");
403        assert_eq!(
404            ctx.render_template("{{ 'hello world' | capitalize }}"),
405            "Hello world"
406        );
407        assert_eq!(
408            ctx.render_template("{{ 'HELLO WORLD' | capitalize }}"),
409            "Hello world"
410        );
411        assert_eq!(ctx.render_template("{{ user | truncate(3) }}"), "bob");
412        assert_eq!(
413            ctx.render_template("{{ 'hello world' | truncate(8) }}"),
414            "hello..."
415        );
416        assert_eq!(
417            ctx.render_template("{{ 'hello world' | truncate(8, true) }}"),
418            "hello..."
419        );
420        assert_eq!(
421            ctx.render_template("{{ 'hello world' | truncate(8, false, '***') }}"),
422            "hello***"
423        );
424
425        // Functions
426        assert_eq!(ctx.render_template("{{ length(user) }}"), "3");
427        assert_eq!(ctx.render_template("{{ basename(path) }}"), "file.txt");
428
429        // No substitutions
430        assert_eq!(ctx.render_template("No vars here"), "No vars here");
431    }

433    #[test]
434    fn test_condition_evaluation() {
435        let mut ctx = VariableContext::new();
436        ctx.set(
437            "status".to_string(),
438            serde_yaml::Value::String("ready".to_string()),
439        );
440        ctx.set("enabled".to_string(), serde_yaml::Value::Bool(true));
441        ctx.set("count".to_string(), serde_yaml::Value::Number(42.into()));
442        ctx.set(
443            "items".to_string(),
444            serde_yaml::Value::Sequence(vec![
445                serde_yaml::Value::String("a".to_string()),
446                serde_yaml::Value::String("b".to_string()),
447            ]),
448        );
449
450        // Basic boolean literals
451        assert!(ctx.evaluate_condition("true"));
452        assert!(!ctx.evaluate_condition("false"));
453
454        // Variable comparisons
455        assert!(ctx.evaluate_condition("{{ enabled }} == true"));
456        assert!(!ctx.evaluate_condition("{{ status }} == pending"));
457        assert!(ctx.evaluate_condition("{{ count }} == 42"));
458        assert!(ctx.evaluate_condition("{{ count }} > 40"));
459
460        // Logical operators
461        assert!(ctx.evaluate_condition("{{ enabled }} and {{ count }} > 40"));
462        assert!(ctx.evaluate_condition("{{ status }} == ready or {{ count }} < 10"));
463        assert!(!ctx.evaluate_condition("not {{ enabled }}"));
464
465        // Membership tests - check if variable exists
466        assert!(ctx.evaluate_condition("items is defined"));
467        assert!(!ctx.evaluate_condition("nonexistent is defined"));
468
469        // Defined checks
470        assert!(ctx.evaluate_condition("status is defined"));
471        assert!(!ctx.evaluate_condition("missing_var is defined"));
472        assert!(ctx.evaluate_condition("missing_var is not defined"));
473    }
