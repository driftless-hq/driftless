1//! Include tasks functionality
2//!
3//! This module provides functionality to include external task files
4//! into the current configuration. Allows for modular configuration
5//! and code reuse.
6//!
7//! # Examples
8//!
9//! ## Include a task file
10//!
11//! This example includes tasks from an external file.
12//!
13//! **YAML Format:**
14//! ```yaml
15//! - type: include_tasks
16//!   description: "Include common setup tasks"
17//!   file: common/setup.yml
18//!   when: "setup_required"
19//!   vars:
20//!     app_name: myapp
21//! ```
22//!
23//! **JSON Format:**
24//! ```json
25//! [
26//!   {
27//!     "type": "include_tasks",
28//!     "description": "Include common setup tasks",
29//!     "file": "common/setup.yml",
30//!     "when": "setup_required",
31//!     "vars": {
32//!       "app_name": "myapp"
33//!     }
34//!   }
35//! ]
36//! ```
37//!
38//! **TOML Format:**
39//! ```toml
40//! [[tasks]]
41//! type = "include_tasks"
42//! description = "Include common setup tasks"
43//! file = "common/setup.yml"
44//! when = "setup_required"
45//!
46//! [tasks.vars]
47//! app_name = "myapp"
48//! ```
49
50use serde::{Deserialize, Serialize};
51use std::collections::HashMap;
52
53/// Include tasks task for modular configurations
54#[derive(Debug, Clone, Serialize, Deserialize)]
55pub struct IncludeTasksTask {
56    /// Optional description of what this task does
57    ///
58    /// Human-readable description of the task's purpose. Used for documentation
59    /// and can be displayed in logs or reports.
60    #[serde(skip_serializing_if = "Option::is_none")]
61    pub description: Option<String>,
62
63    /// File to include
64    ///
65    /// Path to the task file to include.
66    pub file: String,
67
68    /// Variable overrides
69    ///
70    /// Variables to pass to the included tasks.
71    #[serde(default)]
72    pub vars: HashMap<String, serde_json::Value>,
73}
