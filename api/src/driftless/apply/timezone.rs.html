91/// System timezone management task
92#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
93pub struct TimezoneTask {
94    /// Optional description of what this task does
95    ///
96    /// Human-readable description of the task's purpose. Used for documentation
97    /// and can be displayed in logs or reports.
98    #[serde(skip_serializing_if = "Option::is_none")]
99    pub description: Option<String>,
100
101    /// Timezone name (e.g., "America/New_York", "UTC")
102    pub name: String,
103}
104
105use anyhow::{Context, Result};
106use std::fs;
107use std::path::Path;
108
109/// Execute a timezone task
110pub async fn execute_timezone_task(task: &TimezoneTask, dry_run: bool) -> Result<()> {
111    // Get current timezone
112    let current_timezone = get_current_timezone()?;
113
114    if current_timezone == task.name {
115        println!("Timezone already set to: {}", task.name);
116        return Ok(());
117    }
118
119    println!(
120        "Changing timezone from '{}' to '{}'",
121        current_timezone, task.name
122    );
123
124    if dry_run {
125        println!("Would set timezone to: {}", task.name);
126    } else {
127        set_timezone(&task.name)?;
128        println!("Set timezone to: {}", task.name);
129    }
130
131    Ok(())
132}
133
134/// Get the current system timezone
135fn get_current_timezone() -> Result<String> {
136    // Try to read from /etc/timezone first (Ubuntu/Debian style)
137    if Path::new("/etc/timezone").exists() {
138        let content =
139            fs::read_to_string("/etc/timezone").with_context(|| "Failed to read /etc/timezone")?;
140        return Ok(content.trim().to_string());
141    }
142
143    // Try to read from /etc/localtime symlink (RedHat/CentOS style)
144    if let Ok(target) = fs::read_link("/etc/localtime") {
145        if let Some(tz_name) = target.to_str() {
146            // Extract timezone name from path like /usr/share/zoneinfo/America/New_York
147            if tz_name.contains("/zoneinfo/") {
148                let parts: Vec<&str> = tz_name.split("/zoneinfo/").collect();
149                if parts.len() == 2 {
150                    return Ok(parts[1].to_string());
151                }
152            }
153        }
154    }
155
156    // Fallback: try timedatectl command
157    let output = std::process::Command::new("timedatectl")
158        .arg("show")
159        .arg("--property=Timezone")
160        .arg("--value")
161        .output()
162        .with_context(|| "Failed to get timezone from timedatectl")?;
163
164    if output.status.success() {
165        let tz = String::from_utf8_lossy(&output.stdout).trim().to_string();
166        if !tz.is_empty() {
167            return Ok(tz);
168        }
169    }
170
171    Err(anyhow::anyhow!("Could not determine current timezone"))
172}
173
174/// Set the system timezone
175fn set_timezone(timezone: &str) -> Result<()> {
176    // Validate timezone exists
177    let zoneinfo_path = format!("/usr/share/zoneinfo/{}", timezone);
178    if !Path::new(&zoneinfo_path).exists() {
179        return Err(anyhow::anyhow!(
180            "Timezone '{}' does not exist in /usr/share/zoneinfo",
181            timezone
182        ));
183    }
184
185    // Try timedatectl first (systemd systems)
186    let timedatectl_result = std::process::Command::new("timedatectl")
187        .arg("set-timezone")
188        .arg(timezone)
189        .status();
190
191    if timedatectl_result.is_ok() && timedatectl_result.unwrap().success() {
192        return Ok(());
193    }
194
195    // Fallback: manual method (Ubuntu/Debian style)
196    // 1. Update /etc/timezone
197    fs::write("/etc/timezone", format!("{}\n", timezone))
198        .with_context(|| "Failed to write /etc/timezone")?;
199
200    // 2. Update /etc/localtime symlink
201    let _ = fs::remove_file("/etc/localtime"); // Ignore error if file doesn't exist
202    std::os::unix::fs::symlink(zoneinfo_path, "/etc/localtime")
203        .with_context(|| "Failed to create /etc/localtime symlink")?;
204
205    Ok(())
206}
