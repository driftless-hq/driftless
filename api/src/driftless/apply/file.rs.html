1//! File task executor
2//!
3//! Handles file operations: create, modify, delete files with proper permissions.
4//!
5//! # Examples
6//!
7//! ## Create a file with content
8//!
9//! This example creates a simple configuration file with specific permissions and ownership.
10//!
11//! **YAML Format:**
12//! ```yaml
13//! - type: file
14//!   description: "Create nginx configuration file"
15//!   path: /etc/nginx/sites-available/default
16//!   state: present
17//!   content: |
18//!     server {
19//!         listen 80;
20//!         root /var/www/html;
21//!         index index.html index.htm;
22//!
23//!         location / {
24//!             try_files $uri $uri/ =404;
25//!         }
26//!     }
27//!   mode: "0644"
28//!   owner: root
29//!   group: root
30//! ```
31//!
32//! **JSON Format:**
33//! ```json
34//! {
35//!   "type": "file",
36//!   "description": "Create nginx configuration file",
37//!   "path": "/etc/nginx/sites-available/default",
38//!   "state": "present",
39//!   "content": "server {\n    listen 80;\n    root /var/www/html;\n    index index.html index.htm;\n\n    location / {\n        try_files $uri $uri/ =404;\n    }\n}",
40//!   "mode": "0644",
41//!   "owner": "root",
42//!   "group": "root"
43//! }
44//! ```
45//!
46//! **TOML Format:**
47//! ```toml
48//! [[tasks]]
49//! type = "file"
50//! description = "Create nginx configuration file"
51//! path = "/etc/nginx/sites-available/default"
52//! state = "present"
53//! content = """
54//! server {
55//!     listen 80;
56//!     root /var/www/html;
57//!     index index.html index.htm;
58//!
59//!     location / {
60//!         try_files $uri $uri/ =404;
61//!     }
62//! }
63//! """
64//! mode = "0644"
65//! owner = "root"
66//! group = "root"
67//! ```
68//!
69//! ## Copy a file from source
70//!
71//! This example copies an existing file to a new location.
72//!
73//! **YAML Format:**
74//! ```yaml
75//! - type: file
76//!   description: "Copy configuration template"
77//!   path: /etc/myapp/config.yml
78//!   state: present
79//!   source: /opt/myapp/config.template.yml
80//!   mode: "0644"
81//!   owner: myapp
82//!   group: myapp
83//! ```
84//!
85//! ## Remove a file
86//!
87//! This example ensures a file does not exist.
88//!
89//! **YAML Format:**
90//! ```yaml
91//! - type: file
92//!   description: "Remove temporary file"
93//!   path: /tmp/temp-file.txt
94//!   state: absent
95//! ```
96//!
97//! ## Create a directory
98//!
99//! This example creates a directory (note: directories are just files with no content).
100//!
101//! **YAML Format:**
102//! ```yaml
103//! - type: file
104//!   description: "Create application directory"
105//!   path: /opt/myapp
106//!   state: present
107//!   mode: "0755"
108//!   owner: myapp
109//!   group: myapp
110//! ```
111//!

113//!
114//! This example creates a file and registers the result.

146/// File state enumeration
147#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
148#[serde(rename_all = "lowercase")]
149pub enum FileState {
150    /// Ensure file exists
151    Present,
152    /// Ensure file does not exist
153    Absent,
154}
155
156/// File operation task
157///
158/// Manages files and directories - create, modify, or remove files with content,
159/// permissions, and ownership. Similar to Ansible's `file` module.
160#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
161pub struct FileTask {
162    /// Optional description of what this task does
163    ///
164    /// Human-readable description of the task's purpose. Used for documentation
165    /// and can be displayed in logs or reports.
166    #[serde(skip_serializing_if = "Option::is_none")]
167    pub description: Option<String>,
168
169    /// Path to the file or directory
170    ///
171    /// Absolute path to the file or directory to manage.
172    /// Parent directories will not be created automatically.
173    pub path: String,
174
175    /// File state (present, absent)
176    ///
177    /// - `present`: Ensure the file exists with specified properties
178    /// - `absent`: Ensure the file does not exist
179    pub state: FileState,
180
181    /// File content (for present state)
182    ///
183    /// Content to write to the file when state is `present`.
184    /// Mutually exclusive with `source`.
185    #[serde(skip_serializing_if = "Option::is_none")]
186    pub content: Option<String>,
187
188    /// Source file to copy from (alternative to content)
189    ///
190    /// Path to a source file to copy content from when state is `present`.
191    /// Mutually exclusive with `content`.
192    #[serde(skip_serializing_if = "Option::is_none")]
193    pub source: Option<String>,
194
195    /// File permissions (octal string like "0644")
196    ///
197    /// File permissions in octal notation (e.g., "0644", "0755").
198    /// Only applied when creating or modifying files.
199    #[serde(skip_serializing_if = "Option::is_none")]
200    pub mode: Option<String>,
201
202    /// File owner username
203    ///
204    /// Username of the file owner. Only applied when creating or modifying files.
205    #[serde(skip_serializing_if = "Option::is_none")]
206    pub owner: Option<String>,
207
208    /// File group name
209    ///
210    /// Group name for the file. Only applied when creating or modifying files.
211    #[serde(skip_serializing_if = "Option::is_none")]
212    pub group: Option<String>,
213}

220/// Execute a file task
221pub async fn execute_file_task(task: &FileTask, dry_run: bool) -> Result<()> {
222    let path = Path::new(&task.path);
223
224    match task.state {
225        FileState::Present => ensure_file_present(path, task, dry_run).await,
226        FileState::Absent => ensure_file_absent(path, dry_run).await,
227    }
228}

230/// Ensure a file exists with the correct content and permissions
231async fn ensure_file_present(path: &Path, task: &FileTask, dry_run: bool) -> Result<()> {
232    let exists = path.exists();
233
234    if !exists {
235        if dry_run {
236            println!("Would create file: {}", path.display());
237        } else {
238            // Create parent directories if needed
239            if let Some(parent) = path.parent() {
240                fs::create_dir_all(parent).with_context(|| {
241                    format!("Failed to create parent directories for {}", path.display())
242                })?;
243            }
244
245            // Create the file
246            fs::write(path, task.content.as_deref().unwrap_or(""))
247                .with_context(|| format!("Failed to create file {}", path.display()))?;
248            println!("Created file: {}", path.display());
249        }
250    } else {
251        // File exists, check if content needs updating
252        if let Some(content) = &task.content {
253            let current_content = fs::read_to_string(path)
254                .with_context(|| format!("Failed to read existing file {}", path.display()))?;
255
256            if current_content != *content {
257                if dry_run {
258                    println!("Would update content of file: {}", path.display());
259                } else {
260                    fs::write(path, content)
261                        .with_context(|| format!("Failed to update file {}", path.display()))?;
262                    println!("Updated content of file: {}", path.display());
263                }
264            }
265        }
266
267        // Check if source file should be copied
268        if let Some(source) = &task.source {
269            let source_path = Path::new(source);
270            if source_path.exists() {
271                let source_content = fs::read(source_path)
272                    .with_context(|| format!("Failed to read source file {}", source))?;
273
274                let current_content = fs::read(path)
275                    .with_context(|| format!("Failed to read target file {}", path.display()))?;
276
277                if source_content != current_content {
278                    if dry_run {
279                        println!("Would copy {} to {}", source, path.display());
280                    } else {
281                        fs::write(path, source_content).with_context(|| {
282                            format!("Failed to copy {} to {}", source, path.display())
283                        })?;
284                        println!("Copied {} to {}", source, path.display());
285                    }
286                }
287            } else {
288                return Err(anyhow::anyhow!("Source file does not exist: {}", source));
289            }
290        }
291    }
292
293    // Set permissions if specified
294    if let Some(mode) = &task.mode {
295        set_file_permissions(path, mode, dry_run)?;
296    }
297
298    // Set ownership if specified
299    if task.owner.is_some() || task.group.is_some() {
300        set_file_ownership(path, task.owner.as_deref(), task.group.as_deref(), dry_run)?;
301    }
302
303    Ok(())
304}

306/// Ensure a file does not exist
307async fn ensure_file_absent(path: &Path, dry_run: bool) -> Result<()> {
308    if path.exists() {
309        if dry_run {
310            println!("Would remove file: {}", path.display());
311        } else if path.is_file() {
312            fs::remove_file(path)
313                .with_context(|| format!("Failed to remove file {}", path.display()))?;
314            println!("Removed file: {}", path.display());
315        } else {
316            return Err(anyhow::anyhow!(
317                "Path exists but is not a file: {}",
318                path.display()
319            ));
320        }
321    }
322
323    Ok(())
324}

326/// Set file permissions
327fn set_file_permissions(path: &Path, mode: &str, dry_run: bool) -> Result<()> {
328    let mode_u32 = u32::from_str_radix(mode.trim_start_matches("0o"), 8)
329        .with_context(|| format!("Invalid octal mode: {}", mode))?;
330
331    if dry_run {
332        println!("Would set permissions of {} to {}", path.display(), mode);
333    } else {
334        let mut perms = fs::metadata(path)
335            .with_context(|| format!("Failed to get metadata for {}", path.display()))?
336            .permissions();
337
338        perms.set_mode(mode_u32);
339        fs::set_permissions(path, perms)
340            .with_context(|| format!("Failed to set permissions on {}", path.display()))?;
341
342        println!("Set permissions of {} to {}", path.display(), mode);
343    }
344
345    Ok(())
346}
